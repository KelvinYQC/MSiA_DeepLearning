def extract_race(file_name):
    match = re.match(r"\d+_\d+_(\d+)_\d+\.jpg\.chip", file_name)
    if match:
        race = int(match.group(1))
        return race
    else:
        return None
    
def extract_gender(file_name):
    # Extract gender information using regular expressions
    pattern = r'_(\d)_'
    match = re.search(pattern, file_name)
    if match:
        return int(match.group(1))
    else:
        return None

    if file_name.endswith(".jpg.chip.jpg"):
        file_path = os.path.join(folder_path, file_name)
        race = extract_race(file_name)
        gender = extract_gender(file_name)
        if race is not None and gender is not None:
            file_paths.append(file_path)
            labels.append(race)
            gender_labels.append(gender)
            if gender == 0:
                male_index.append(index)
            else:
                female_index.append(index)

def preprocess_image(file_path):
    # Load and preprocess the image
    image = tf.io.read_file(file_path)
    image = tf.image.decode_jpeg(image, channels=3)
    image = tf.image.resize(image, [224, 224])  # Adjust the image size as needed
    image = image / 255.0  # Normalize the pixel values to the range [0, 1]
    return image

model = tf.keras.Sequential([
    tf.keras.applications.MobileNetV2(input_shape=(224, 224, 3), include_top=False),
    tf.keras.layers.GlobalAveragePooling2D(),
    tf.keras.layers.Dense(256, activation='relu'),
    tf.keras.layers.Dropout(0.5),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dropout(0.5),
    tf.keras.layers.Dense(num_classes, activation='softmax') 
])

MobileNetV2: The MobileNetV2 layer is used as the base convolutional neural network (CNN) model. It is a lightweight and efficient CNN architecture that has been pre-trained on a large dataset (ImageNet). By including this layer, your model can leverage the powerful feature extraction capabilities learned from ImageNet. It can help us to extract details form the face picture data.

Based on the pretrained layers, we add some more layers.

GlobalAveragePooling2D: This layer reduces the spatial dimensions of the input features from the previous layer. It computes the average value for each feature map, resulting in a fixed-length vector that summarizes the important features of the input.

Dense: The first Dense layer consists of 256 neurons and uses the ReLU activation function. It introduces non-linearity to the model and allows for learning complex patterns in the data.

Dropout: The first Dropout layer is added after the first Dense layer. It randomly sets a fraction of input units to 0 during training, which helps prevent overfitting by providing regularization and reducing the reliance on specific features.

Dense: The second Dense layer consists of 128 neurons and also uses the ReLU activation function. It further enhances the model's capacity to learn intricate patterns and relationships in the data.

Dropout: The second Dropout layer is added after the second Dense layer. It serves the same purpose as the first Dropout layer by reducing overfitting and improving generalization.

Dense: The final Dense layer consists of 5 neural corresponding to the 5 different classes of outputs. They have an activation function of softmax.
